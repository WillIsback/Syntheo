AI API Reference

All DGX Spark AI services exposed through a single authenticated gateway. One API key, five services.

Base URL https://api.willisback.fr/{service}/v1
Keys https://apikeys.willisback.fr  → create & manage your keys
Language:
🔑

Authentication

All requests require a Bearer token in the Authorization header. Get your key at apikeys.willisback.fr.

ℹ️

The gateway translates your key to the correct upstream service key automatically — you never need the raw DGX keys.

# Every request needs this header
Authorization: Bearer sk-xxxxxxxxxxxxxxxxxxxx
⚡

Quickstart

Install the OpenAI SDK — it works for LLM, embedding, and OCR out of the box.

# pip install openai httpx
from openai import OpenAI

client = OpenAI(
    base_url="https://api.willisback.fr/llm/v1",
    api_key="sk-your-key-here",
)

resp = client.chat.completions.create(
    model="nvidia/Qwen3.6-35B-A3B-NVFP4",
    messages=[{"role": "user", "content": "Hello!"}],
    extra_body={"chat_template_kwargs": {"enable_thinking": False}},
)
print(resp.choices[0].message.content)
// npm install openai
import OpenAI from "openai";

const client = new OpenAI({
  baseURL: "https://api.willisback.fr/llm/v1",
  apiKey:  "sk-your-key-here",
});

const resp = await client.chat.completions.create({
  model:    "nvidia/Qwen3.6-35B-A3B-NVFP4",
  messages: [{ role: "user", content: "Hello!" }]
});
console.log(resp.choices[0].message.content);

LLM · Qwen3-35B

/llm/v1

OpenAI-compatible chat completions via vLLM. Supports streaming, system prompts, and extended thinking mode.

ParameterTypeDescription
modelstringrequirednvidia/Qwen3.6-35B-A3B-NVFP4
messagesarrayrequiredConversation history
streambooleanoptionalServer-sent events streaming
max_tokensintegeroptionalMax completion tokens
temperaturefloatoptional0 = deterministic, 1 = creative
enable_thinkingbooleanoptionalChain-of-thought (slower, more accurate)

Basic chat

from openai import OpenAI

llm = OpenAI(
    base_url="https://api.willisback.fr/llm/v1",
    api_key="sk-your-key",
)

resp = llm.chat.completions.create(
    model="nvidia/Qwen3.6-35B-A3B-NVFP4",
    messages=[
        {"role": "system", "content": "You are a helpful assistant."},
        {"role": "user",   "content": "Explain RAG in 2 sentences."},
    ],
    max_tokens=256,
    temperature=0.7,
    extra_body={"chat_template_kwargs": {"enable_thinking": False}},
)
print(resp.choices[0].message.content)
import OpenAI from "openai";

const llm = new OpenAI({
  baseURL: "https://api.willisback.fr/llm/v1",
  apiKey:  "sk-your-key",
});

const resp = await llm.chat.completions.create({
  model:       "nvidia/Qwen3.6-35B-A3B-NVFP4",
  messages: [
    { role: "system", content: "You are a helpful assistant." },
    { role: "user",   content: "Explain RAG in 2 sentences."   },
  ],
  max_tokens:  256,
  temperature: 0.7,
});
console.log(resp.choices[0].message.content);

Streaming

with llm.chat.completions.stream(
    model="nvidia/Qwen3.6-35B-A3B-NVFP4",
    messages=[{"role": "user", "content": "Write a haiku about AI."}],
    extra_body={"chat_template_kwargs": {"enable_thinking": False}},
) as stream:
    for chunk in stream:
        print(chunk.choices[0].delta.content or "", end="", flush=True)
const stream = await llm.chat.completions.create({
  model:    "nvidia/Qwen3.6-35B-A3B-NVFP4",
  messages: [{ role: "user", content: "Write a haiku about AI." }],
  stream:   true,
});
for await (const chunk of stream) {
  process.stdout.write(chunk.choices[0]?.delta?.content ?? "");
}

Thinking mode (chain-of-thought)

# enable_thinking=True → slower but more accurate on complex tasks
# extra_body merges the dict into the request body — vLLM reads it
resp = llm.chat.completions.create(
    model="nvidia/Qwen3.6-35B-A3B-NVFP4",
    messages=[{"role": "user", "content": "Solve: 17 × 23 step by step"}],
    extra_body={"chat_template_kwargs": {"enable_thinking": True}},
)
# content holds the final answer; reasoning holds the thinking trace
print(resp.choices[0].message.content)
// JS SDK: pass vLLM-specific params with "as any" to bypass TS types
const resp = await llm.chat.completions.create({
  model:    "nvidia/Qwen3.6-35B-A3B-NVFP4",
  messages: [{ role: "user", content: "Solve: 17 × 23 step by step" }],
  chat_template_kwargs: { enable_thinking: true },  // vLLM extension
} as any);
console.log(resp.choices[0].message.content);

Embedding · BGE-M3

/embed/v1

Multilingual dense embeddings (1024 dimensions) via BAAI/bge-m3. Works with the OpenAI SDK embeddings.create().

ParameterTypeDescription
modelstringrequiredbge-m3
inputstring | string[]requiredText(s) to embed — batch up to 32
from openai import OpenAI

emb = OpenAI(
    base_url="https://api.willisback.fr/embed/v1",
    api_key="sk-your-key",
)

resp = emb.embeddings.create(
    model="bge-m3",
    input=[
        "What is retrieval-augmented generation?",
        "RAG combines retrieval with generation.",
    ],
)

for item in resp.data:
    print(f"index {item.index}: dim={len(item.embedding)}")
# → index 0: dim=1024
# → index 1: dim=1024
import OpenAI from "openai";

const emb = new OpenAI({
  baseURL: "https://api.willisback.fr/embed/v1",
  apiKey:  "sk-your-key",
});

const resp = await emb.embeddings.create({
  model: "bge-m3",
  input: [
    "What is retrieval-augmented generation?",
    "RAG combines retrieval with generation.",
  ],
});

resp.data.forEach(item =>
  console.log(`index ${item.index}: dim=${item.embedding.length}`)
);

Reranking · BGE-Reranker

/rerank/v1

Cross-encoder reranking — score and reorder documents by relevance to a query. Not part of the OpenAI SDK; use httpx / fetch directly.

ParameterTypeDescription
modelstringrequiredbge-reranker
querystringrequiredThe search query
documentsstring[]requiredDocuments to rank
top_nintegeroptionalReturn top N results (default: all)
import httpx

resp = httpx.post(
    "https://api.willisback.fr/rerank/v1/rerank",
    headers={"Authorization": "Bearer sk-your-key"},
    json={
        "model":     "bge-reranker",
        "query":     "What is the capital of France?",
        "documents": [
            "Paris is the capital of France.",
            "Berlin is the capital of Germany.",
            "Madrid is the capital of Spain.",
        ],
        "top_n": 3,
    },
    timeout=15,
)

for result in resp.json()["results"]:
    print(f"{result['relevance_score']:.4f}  {result['document']['text']}")
# → 0.9821  Paris is the capital of France.
# → 0.0043  Berlin is the capital of Germany.
const resp = await fetch("https://api.willisback.fr/rerank/v1/rerank", {
  method:  "POST",
  headers: {
    "Authorization": "Bearer sk-your-key",
    "Content-Type":  "application/json",
  },
  body: JSON.stringify({
    model:     "bge-reranker",
    query:     "What is the capital of France?",
    documents: [
      "Paris is the capital of France.",
      "Berlin is the capital of Germany.",
    ],
    top_n: 2,
  }),
});

const data = await resp.json();
data.results.forEach(r =>
  console.log(r.relevance_score.toFixed(4), r.document.text)
);

OCR · LightOn

/ocr/v1

Vision-based OCR via LightOn. Send an image as a base64 data URL or HTTP URL and receive the extracted text. Uses the OpenAI vision message format.

ParameterTypeDescription
modelstringrequiredlighton-ocr
image_url.urlstringrequiredHTTP URL or data:image/...;base64,...
max_tokensintegeroptionalMax output tokens (default 512)
import base64
from pathlib import Path
from openai import OpenAI

ocr = OpenAI(
    base_url="https://api.willisback.fr/ocr/v1",
    api_key="sk-your-key",
)

# From local file
img_b64 = base64.b64encode(Path("document.png").read_bytes()).decode()

resp = ocr.chat.completions.create(
    model="lighton-ocr",
    messages=[{
        "role": "user",
        "content": [
            {"type": "text", "text": "Transcribe all text in this image."},
            {"type": "image_url", "image_url": {
                "url": f"data:image/png;base64,{img_b64}"
            }},
        ],
    }],
    max_tokens=1024,
)

print(resp.choices[0].message.content)
import OpenAI from "openai";
import fs     from "fs";

const ocr = new OpenAI({
  baseURL: "https://api.willisback.fr/ocr/v1",
  apiKey:  "sk-your-key",
});

// From local file (Node.js)
const b64 = fs.readFileSync("document.png").toString("base64");

const resp = await ocr.chat.completions.create({
  model:    "lighton-ocr",
  messages: [{
    role:    "user",
    content: [
      { type: "text",      text: "Transcribe all text in this image." },
      { type: "image_url", image_url: { url: `data:image/png;base64,${b64}` }},
    ],
  }],
  max_tokens: 1024,
});
console.log(resp.choices[0].message.content);

WhisperX

/whisper/asr/v1

Async transcription with speaker diarization. Submit a job → receive a job_id → poll until done. Also exports SRT / VTT / TXT subtitle files.

Endpoints

MethodPathDescription
POST/whisper/asr/v1/transcribeSubmit audio → returns job_id (202)
GET/whisper/asr/v1/jobs/{job_id}Poll status + full JSON result
GET/whisper/asr/v1/jobs/{job_id}/srtDownload SRT subtitle file
GET/whisper/asr/v1/jobs/{job_id}/vttDownload VTT subtitle file
GET/whisper/asr/v1/jobs/{job_id}/txtDownload plain text transcript

POST /transcribe — multipart/form-data fields

FieldTypeDescription
audio_filebinaryrequiredAudio file (mp3, wav, m4a, ogg, webm, flac)
languagestringoptionalISO-639-1 code — default fr
initial_promptstringoptionalContext hint to guide transcription quality
hotwordsstringoptionalComma-separated keywords to favour
num_speakersintegeroptionalExpected speaker count (improves diarization)
import httpx, time

BASE = "https://api.willisback.fr/whisper/asr/v1"
HDR  = {"Authorization": "Bearer sk-your-key"}

# 1 — Submit job (returns immediately with job_id)
with open("meeting.mp3", "rb") as f:
    r = httpx.post(
        f"{BASE}/transcribe",
        headers=HDR,
        files={"audio_file": ("meeting.mp3", f, "audio/mpeg")},
        data={"language": "fr", "num_speakers": 2},
        timeout=30,
    )
job_id = r.json()["job_id"]
print(f"Submitted: {job_id}")

# 2 — Poll until completed (transcription + alignment + diarization run server-side)
while True:
    result = httpx.get(f"{BASE}/jobs/{job_id}", headers=HDR).json()
    if result["status"] in ("completed", "failed"):
        break
    time.sleep(3)

# 3 — Per-speaker segments (no top-level text field — use segments)
for seg in result["segments"]:
    print(f"[{seg['speaker']}] {seg['start']:.1f}s–{seg['end']:.1f}s  {seg['text'].strip()}")
# → [SPEAKER_00]  0.6s–8.6s   Ce que nous appelons grippe tropicale…
# → [SPEAKER_00]  8.7s–19.9s  La dengue est un arbovirus…

# Assemble full transcript from segments
full_text = " ".join(s["text"].strip() for s in result["segments"])

# 4 — Download SRT subtitles
from pathlib import Path
srt = httpx.get(f"{BASE}/jobs/{job_id}/srt", headers=HDR).text
Path("meeting.srt").write_text(srt)
import fs from "fs";

const BASE = "https://api.willisback.fr/whisper/asr/v1";
const HDR  = { Authorization: "Bearer sk-your-key" };

// 1 — Submit job
const form = new FormData();
form.append("audio_file",
  new Blob([fs.readFileSync("meeting.mp3")], { type: "audio/mpeg" }),
  "meeting.mp3"
);
form.append("language",     "fr");
form.append("num_speakers", "2");

const { job_id } = await (await fetch(`${BASE}/transcribe`, {
  method: "POST", headers: HDR, body: form,
})).json();

// 2 — Poll until done
let result;
do {
  await new Promise(r => setTimeout(r, 3000));
  result = await (await fetch(`${BASE}/jobs/${job_id}`, { headers: HDR })).json();
} while (!["completed", "failed"].includes(result.status));

// 3 — Per-speaker segments (no top-level text field — use segments)
result.segments.forEach(seg =>
  console.log(`[${seg.speaker}] ${seg.start.toFixed(1)}s–${seg.end.toFixed(1)}s  ${seg.text.trim()}`)
);

// Assemble full transcript
const fullText = result.segments.map(s => s.text.trim()).join(" ");

// 4 — Download SRT
const srt = await (await fetch(`${BASE}/jobs/${job_id}/srt`, { headers: HDR })).text();
fs.writeFileSync("meeting.srt", srt);

Error codes

HTTPCodeCause
401invalid_api_keyMissing, wrong, or revoked key
429rate_limit_exceededOver 60 requests / minute per key
502—DGX service temporarily unavailable
504—DGX inference timeout (long audio / large prompt)
from openai import OpenAI, AuthenticationError, RateLimitError

try:
    resp = client.chat.completions.create(...)
except AuthenticationError:
    print("Invalid or revoked key")
except RateLimitError:
    print("Rate limit hit — wait and retry")
try {
  const resp = await client.chat.completions.create(...);
} catch (e) {
  if (e.status === 401) console.error("Invalid or revoked key");
  if (e.status === 429) console.error("Rate limit — wait and retry");
}
